# record video and save its frames - IMPORTANT press q to stop
ros2 run image_capture image_capture --video y


### other image capture functions
# take photo automatically (1 per sec, e.g, right camera):
ros2 run image_capture image_capture 

# take photo upon pressing 's' only (e.g., left camera):
ros2 run image_capture image_capture --key_press y --camera left
